 제이쿼리는 자바스크립트의 기능을 쉽고 원할하게 사용할 수 있는 오픈 소스 기반 자바스크립트의 라이브러리입니다. 제이쿼리를 사용했을 경우, 바닐라 자바스크립트에 비해 코드의 길이가 짧아지고 시간 또한 절약이 된다는 장점이 있지만 저는 제이쿼리를 꺼려하는 편입니다. 제이쿼리 자체가 자바스크립트에 비해 정확성이 약간 떨어지고 버전마다 지원하지 않는 기능도 있기 떄문에 크로스 브라우징에 있어 치명타를 입는 것 같습니다. 무엇보다 더 치명적인 단점은, 사용할수록 랜딩 시간이 길어지는 개떡같은 경우가 종종 발생한다는 것이지요... 이런 저런 이유로 저는 조별 프로젝트 때 제이쿼리로 크게 데였기 때문에 바닐라 자바스크립트를 만을 활용하여 코딩하고 있습니다.

### 플러그인보다 손으로 코딩하는걸 좋아해요
 ```JavaScript
$(window).on("load", function () {
  function fade() {
    let animation_height = $(window).innerHeight() * 0.5;
    let ratio = Math.round((1 / animation_height) * 10000) / 10000;
      $(".fadeB").each(function () {
      let objectTop = $(this).offset().top;
      let windowBottom = $(window).scrollTop() + $(window).innerHeight();
      if (objectTop < windowBottom) {
      if (objectTop < windowBottom - animation_height) {
          $(this).css({
              transition: "opacity 1s ease-out",
              transition: "top 1s ease-out",
              opacity: 1,
              top: "0px",
          });
          } else {
          $(this).css({
              transition: "opacity 1s ease-out",
              opacity: (windowBottom - objectTop) * ratio,
              transition: "top 1s ease-out",
              top: `${600 * (1 - (windowBottom - objectTop) * ratio)}px`,
          });
          }
      } else {
          $(this).css({
          opacity: 0,
          top: "600px",
          });
        }
    }); 
  }

  $(".fadeB").css({
      opacity: 0,
      top: "600px",
  });
  fade();

  $(window).scroll(function () {
      fade();
  });
});
 ```
